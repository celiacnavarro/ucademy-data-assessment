Script de Preprocesamiento de Datos de Marketing

Este script limpia y prepara datos de marketing (campañas, inscripciones y leads) para su análisis posterior. Genera:

  • Datos limpios.
  • Estadísticas resumen.
  • Una base de datos SQLite reutilizable.
  • Archivos CSV de respaldo.
  • Un reporte en formato JSON y TXT con estadísticas.

Estructura:

  1. Configuración de rutas.
  2. Carga de datos originales.
  3. Limpieza y transformación.
  4. Categorización de fuentes de leads.
  5. Creación de estadísticas resumen.
  6. Guardado en base de datos y archivos.

Outputs:

  • CSVs con los datos preprocesados.
  • Base de datos 'marketing_analysis.db'
  • JSON y TXT con resumen exploratorio de los datos.

 78# 3. Limpieza de datos financieros

 80def clean_financial_data(df, column):
 81    """
 82    Limpia columnas financieras eliminando símbolos (€) y convirtiendo comas a puntos.
 83    Convierte el resultado a tipo float.
 84
 85    Parámetros:
 86        df (pandas.DataFrame): DataFrame que contiene la columna.
 87        column (str): Nombre de la columna a limpiar.
 88
 89    Retorna:
 90        Series (float): Columna limpia y convertida a float.
 91    """
 92    return (df[column]
 93            .str.replace("€", "", regex=False)
 94            .str.replace(",", ".", regex=False)
 95            .astype(float))

100def clean_campaigns(campaigns):
101    """
102    Limpia y transforma la tabla de campañas.
103
104    Transformaciones:
105    - Limpia costos financieros.
106    - Normaliza texto.
107    - Extrae tipo de curso y canal desde el nombre.
108    - Convierte fechas a datetime.
109
110    Parámetros:
111        campaigns (pandas.DataFrame): Tabla original de campañas.
112
113    Retorna:
114        campaigns (pandas.DataFrame): Tabla limpia y transformada.
115    """
116    print("Limpiando datos de campañas...")
117
118    campaigns['cost'] = clean_financial_data(campaigns, 'cost')
119    campaigns["name"] = campaigns["name"].str.lower().str.strip()
120    campaigns["campaign_id"] = campaigns["campaign_id"].str.lower().str.strip()
121
122    # Extraer course_type y channel
123    campaigns["course_type"] = campaigns["name"].apply(
124        lambda x: x.split("_")[0] if "_" in x else "unknown"
125    )
126    campaigns["channel"] = campaigns["name"].apply(
127        lambda x: x.split("_")[1] if "_" in x else "unknown"
128    )
129
130    # Convertir fechas
131    campaigns['started_at'] = pd.to_datetime(campaigns['started_at'])
132    campaigns['ended_at'] = pd.to_datetime(campaigns['ended_at'])
133
134    print(f"Procesadas {len(campaigns)} campañas")
135    return campaigns

182def categorize_all_lead_sources(leads, campaigns):
183    """
184    Categorización de leads según su fuente.
185
186    Clasificación:
187    - paid_campaign: proveniente de una campaña pagada.
188    - organic: proveniente de búsqueda orgánica.
189    - partner: proveniente de partners/referrals.
190    - other: cualquier otro caso.
191
192    Parámetros:
193        leads (pandas.DataFrame): Tabla de leads.
194        campaigns (pandas.DataFrame): Tabla de campañas limpias.
195
196    Retorna:
197        leads (pandas.DataFrame): Leads con columnas adicionales de categorización.
198    """
199    print("Categorizando fuentes de leads...")
200
201    campaign_mapping = campaigns.set_index('campaign_id')[['channel', 'course_type']].to_dict('index')
202
203    def categorize_source(input_channel):
204        if pd.isna(input_channel) or input_channel == '' or input_channel == 'null':
205            return 'unknown', 'unknown', 'unknown'
206        input_channel_lower = str(input_channel).lower().strip()
207
208        if input_channel_lower in campaign_mapping:
209            campaign_info = campaign_mapping[input_channel_lower]
210            return 'paid_campaign', campaign_info['channel'], campaign_info['course_type']
211        elif 'organic' in input_channel_lower or 'web_organic' in input_channel_lower:
212            return 'organic', 'web', 'unknown'
213        elif 'partner' in input_channel_lower or 'ref' in input_channel_lower:
214            return 'partner', 'referral', 'unknown'
215        else:
216            return 'other', 'unknown', 'unknown'
217
218    categorization_results = leads['input_channel'].apply(categorize_source)
219    leads['source_category'] = [result[0] for result in categorization_results]
220    leads['channel_derived'] = [result[1] for result in categorization_results]
221    leads['course_type_derived'] = [result[2] for result in categorization_results]
222
223    print("Distribución de categorías de fuentes:")
224    print(leads['source_category'].value_counts())
225
226    return leads

231def create_summary_stats(campaigns, leads, inscriptions):
232    """
233    Genera estadísticas generales de los datos procesados.
234
235    Parámetros:
236        campaigns (pandas.DataFrame): Campañas limpias.
237        leads (pandas.DataFrame): Leads limpios.
238        inscriptions (pandas.DataFrame): Inscripciones limpias.
239
240    Retorna:
241        stats (dict): Estadísticas resumidas.
242    """
243    print("Generando estadísticas resumen...")
244    stats = {
245        'data_overview': {
246            'total_campaigns': len(campaigns),
247            'total_leads': len(leads),
248            'total_inscriptions': len(inscriptions),
249            'date_range_leads': f"{leads['created_at'].min()} to {leads['created_at'].max()}",
250            'date_range_inscriptions': f"{inscriptions['created_at'].min()} to {inscriptions['created_at'].max()}",
251        },
252        'lead_sources': leads['source_category'].value_counts().to_dict(),
253        'campaign_channels': campaigns['channel'].value_counts().to_dict(),
254        'course_types': campaigns['course_type'].value_counts().to_dict(),
255        'processing_date': datetime.now().isoformat()
256    }
257    return stats

262def save_to_database(campaigns, leads, inscriptions, db_path):
263    """
264    Guarda los DataFrames limpios en una base de datos SQLite.
265
266    Este procedimiento persiste las tablas procesadas en un fichero SQLite para
267    facilitar consultas posteriores desde los scripts de análisis. Las tablas se
268    sobrescriben en cada ejecución (if_exists="replace").
269
270    Parámetros:
271        campaigns (pandas.DataFrame): DataFrame con la tabla `campaigns` ya limpiada.
272        leads (pandas.DataFrame): DataFrame con la tabla `leads` ya limpiada.
273        inscriptions (pandas.DataFrame): DataFrame con la tabla `inscriptions` ya limpiada.
274        db_path (str): Ruta completa al fichero de base de datos SQLite donde se guardarán las tablas.
275    
276    Retorna:
277        None
278    
279    Notas:
280    Sobrescribe las tablas 'campaigns', 'leads' e 'inscriptions' en la base de datos SQLite.
281
282    """
283    print("Guardando datos limpios en base de datos SQLite...")
284    conn = sqlite3.connect(db_path)
285    campaigns.to_sql("campaigns", conn, if_exists="replace", index=False)
286    inscriptions.to_sql("inscriptions", conn, if_exists="replace", index=False)
287    leads.to_sql("leads", conn, if_exists="replace", index=False)
288    conn.close()
289    print(f"Base de datos guardada: {db_path}")

315def save_summary_report(stats, processed_data_path):
316    """
317    Guarda un reporte resumen del procesamiento de datos.
318
319    Parámetros:
320    stats : dict
321        Diccionario con estadísticas y métricas generadas durante el procesamiento.
322        Debe contener al menos:
323            - 'processing_date': str, fecha del procesamiento.
324            - 'data_overview': dict con totales y rangos de fechas.
325            - 'lead_sources': dict con conteos por fuente de lead.
326            - 'campaign_channels': dict con conteos por canal.
327            - 'course_types': dict con conteos por tipo de curso.
328    processed_data_path : str
329        Ruta donde se guardarán los reportes generados.
330
331    Retorna:
332    report : str
333        Texto del reporte generado, útil para imprimir en consola.
334    
335    Notas:
336    Genera dos archivos en processed_data_path:
337    - data_summary_<timestamp>.json
338    - data_summary_<timestamp>.txt
339
340    """
341    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
342
343    # Guardar JSON de estadísticas
344    with open(os.path.join(processed_data_path, f'data_summary_{timestamp}.json'), 'w') as f:
345        json.dump(stats, f, indent=2, default=str)
346
347    # Generar reporte TXT legible
348    report = f"""
349RESUMEN DE PROCESAMIENTO DE DATOS
350Fecha de Procesamiento: {stats['processing_date']}
351
352RESUMEN DE DATOS:
353Total Campañas: {stats['data_overview']['total_campaigns']:,}
354Total Leads: {stats['data_overview']['total_leads']:,}
355Total Inscripciones: {stats['data_overview']['total_inscriptions']:,}
356Rango Fechas Leads: {stats['data_overview']['date_range_leads']}
357Rango Fechas Inscripciones: {stats['data_overview']['date_range_inscriptions']}
358
359BREAKDOWN POR FUENTES DE LEADS:
360"""
361    for source, count in stats['lead_sources'].items():
362        percentage = (count / stats['data_overview']['total_leads']) * 100
363        report += f"• {source.title()}: {count:,} ({percentage:.1f}%)\n"
364
365    report += f"\nCANALES DE CAMPAÑA:\n"
366    for channel, count in stats['campaign_channels'].items():
367        report += f"• {channel.title()}: {count} campañas\n"
368
369    report += f"\nTIPOS DE CURSO:\n"
370    for course, count in stats['course_types'].items():
371        report += f"• {course.title()}: {count} campañas\n"
372
373    # Guardar reporte TXT
374    with open(os.path.join(processed_data_path, f'data_summary_{timestamp}.txt'), 'w') as f:
375        f.write(report)
376
377    print(f"Reporte resumen guardado en {processed_data_path}")
378    return report

381def main():
382    """
383    Función principal de preprocesamiento de datos.
384
385    Flujo de trabajo:
386    1. Configura rutas de datos.
387    2. Carga datasets originales: campañas, inscripciones, leads.
388    3. Limpia los datos cargados.
389    4. Categoriza leads según canal de origen.
390    5. Crea estadísticas resumen.
391    6. Guarda datos en base de datos.
392    7. Guarda copias procesadas en CSV.
393    8. Genera y guarda reporte resumen.
394
395    Retorna:
396    db_path : str
397        Ruta donde se guardó la base de datos procesada.
398    processed_data_path : str
399        Ruta donde se guardaron los datos procesados y reportes.
400    """
401    print("Iniciando pipeline de preprocesamiento de datos...")
402
403    # 1. Configurar rutas
404    data_path, processed_data_path, db_path = setup_paths()
405
406    # 2. Cargar datos originales
407    campaigns, inscriptions, leads = load_raw_data(data_path)
408
409    # 3. Limpiar datos
410    campaigns_clean = clean_campaigns(campaigns)
411    inscriptions_clean = clean_inscriptions(inscriptions)
412    leads_clean = clean_leads(leads)
413
414    # 4. Categorizar leads
415    leads_clean = categorize_all_lead_sources(leads_clean, campaigns_clean)
416
417    # 5. Crear estadísticas
418    stats = create_summary_stats(campaigns_clean, leads_clean, inscriptions_clean)
419
420    # 6. Guardar en base de datos
421    save_to_database(campaigns_clean, leads_clean, inscriptions_clean, db_path)
422
423    # 7. Guardar CSVs de backup
424    save_processed_csvs(campaigns_clean, leads_clean, inscriptions_clean, processed_data_path)
425
426    # 8. Guardar reporte resumen
427    report = save_summary_report(stats, processed_data_path)
428
429    print("\n" + "="*50)
430    print("PREPROCESAMIENTO DE DATOS COMPLETADO!")
431    print("="*50)
432    print(report)
433
434    return db_path, processed_data_path
435
436
437if __name__ == "__main__":
438    db_path, processed_data_path = main()

184    """
185    Categorización de leads según su fuente.
186
187    Clasificación:
188    - paid_campaign: proveniente de una campaña pagada.
189    - organic: proveniente de búsqueda orgánica.
190    - partner: proveniente de partners/referrals.
191    - other: cualquier otro caso.
192
193    Parámetros:
194        leads (pandas.DataFrame): Tabla de leads.
195        campaigns (pandas.DataFrame): Tabla de campañas limpias.
196
197    Retorna:
198        leads (pandas.DataFrame): Leads con columnas adicionales de categorización.
199    """
200    print("Categorizando fuentes de leads...")
201
202    campaign_mapping = campaigns.set_index('campaign_id')[['channel', 'course_type']].to_dict('index')
203
204    def categorize_source(input_channel):
205        if pd.isna(input_channel) or input_channel == '' or input_channel == 'null':
206            return 'unknown', 'unknown', 'unknown'
207        input_channel_lower = str(input_channel).lower().strip()
208
209        if input_channel_lower in campaign_mapping:
210            campaign_info = campaign_mapping[input_channel_lower]
211            return 'paid_campaign', campaign_info['channel'], campaign_info['course_type']
212        elif 'organic' in input_channel_lower or 'web_organic' in input_channel_lower:
213            return 'organic', 'web', 'unknown'
214        elif 'partner' in input_channel_lower or 'ref' in input_channel_lower:
215            return 'partner', 'referral', 'unknown'
216        else:
217            return 'other', 'unknown', 'unknown'
218
219    categorization_results = leads['input_channel'].apply(categorize_source)
220    leads['source_category'] = [result[0] for result in categorization_results]
221    leads['channel_derived'] = [result[1] for result in categorization_results]
222    leads['course_type_derived'] = [result[2] for result in categorization_results]
223
224    print("Distribución de categorías de fuentes:")
225    print(leads['source_category'].value_counts())
226
227    return leads
